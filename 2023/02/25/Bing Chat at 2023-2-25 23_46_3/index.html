Q: 请给我看一下那段代码。

Bing: 好的，这是一段代码12：

(function() {
    var lang = 'zh-CN';
    var url = 'https://raw.githubusercontent.com/Notion-zh-CN/Notion-zh-CN/master/notion_zh_CN.json';
    var xhr = new XMLHttpRequest();
    xhr.open('GET', url);
    xhr.onload = function() {
        if (xhr.status === 200) {
            var data = JSON.parse(xhr.responseText);
            window.notionI18nOverride = data[lang];
            window.notionI18nLocale = lang;
        }
        else {
            console.log('Request failed: ' + xhr.status);
        }
    };
    xhr.send();
})();

您可以将这段代码复制到Console面板中，然后按下回车键，再刷新页面。这样就可以看到Notion的中文界面了。
